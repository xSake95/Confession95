        function runDiagnostics() {

            if (diagnosticsStarted) return;

            diagnosticsStarted = true;

           

            const steps = [

                { id: 'mem-test', text: 'OK - 640K', delay: 500 },

                { id: 'script-test', text: 'OK - 66 BOOKS', delay: 800 },

                { id: 'buffer-test', text: 'OK - 64KB', delay: 1100 },

                { id: 'soul-test', text: 'CONNECTED', delay: 1400 },

                { id: 'grace-test', text: 'UNLIMITED', delay: 1700 }

            ];

           

            let progress = 0;

            const progressFill = document.getElementById('progress-fill');

            const loadingText = document.getElementById('loading-text');

            const enterBtn = document.getElementById('enter-btn');

           

            steps.forEach((step, index) => {

                setTimeout(() => {

                    const el = document.getElementById(step.id);

                    if (el) {

                        el.textContent = step.text;

                        el.classList.add('status');

                    }

                   

                    progress = ((index + 1) / steps.length) * 100;

                    if (progressFill) progressFill.style.width = progress + '%';

                   

                    if (index === steps.length - 1) {

                        if (loadingText) {

                            loadingText.textContent = 'SYSTEM READY. PRESS ENTER TO CONFESS.';

                            loadingText.style.color = '#ff0';

                        }

                        if (enterBtn) enterBtn.disabled = false;

                    }

                }, step.delay);

            });

        }

       

        function skipDiagnostics() {

            document.getElementById('startup').style.display = 'none';

            document.getElementById('main-menu').style.display = 'block';

        function enterSystem() {

            document.getElementById('startup').style.display = 'none';

            document.getElementById('main-menu').style.display = 'block';

        }

       

        function showScreen(screenId) {

            document.getElementById('startup').style.display = 'none';

            document.getElementById('main-menu').style.display = 'none';

            document.querySelectorAll('.content-screen').forEach(s => s.style.display = 'none');

           

            if (screenId === 'main-menu') {

                document.getElementById('main-menu').style.display = 'block';

            } else {

                document.getElementById(screenId).style.display = 'block';

            }

           

            if (screenId === 'saved') {

                renderSavedConfessions();

            }

        }

            }

        }

       

        function selectSin(sin) {

            currentSin = sin;

           

            document.querySelectorAll('.sin-item').forEach(item => {

                item.classList.remove('selected');

            });

            event.target.classList.add('selected');

           

            const verses = sinDatabase[sin];

            const randomVerse = verses[Math.floor(Math.random() * verses.length)];

           

            document.getElementById('verse-text').textContent = '"' + randomVerse.verse + '"';

            document.getElementById('verse-ref').textContent = '— ' + randomVerse.reference;

           

            document.getElementById('confession-area').style.display = 'block';

            document.getElementById('confession-text').value = '';

            document.getElementById('confession-text').focus();

        }

       

        function saveConfession() {

            const text = document.getElementById('confession-text').value.trim();

           

            if (!text) {

                showNotification('ERROR', 'Please type your confession before saving.', true);

                return;

            }

           

            if (!currentSin) {

                showNotification('ERROR', 'No sin selected.', true);

                return;

            }

           

            const confession = {

                id: Date.now(),

                sin: currentSin.toUpperCase(),

                sinKey: currentSin,

                text: text,

                verse: document.getElementById('verse-text').textContent,

                reference: document.getElementById('verse-ref').textContent,

                date: new Date().toLocaleString()

            };

           

            savedConfessions.unshift(confession);

            localStorage.setItem('confessions', JSON.stringify(savedConfessions));

           

            showNotification('GRACE APPLIED', 'Your confession has been saved. Go in peace.');

           

            document.getElementById('confession-text').value = '';

            document.querySelectorAll('.sin-item').forEach(item => {

                item.classList.remove('selected');

            });

            document.getElementById('confession-area').style.display = 'none';

            currentSin = null;

        }

       

        function renderSavedConfessions() {

            const container = document.getElementById('saved-list');

           

            if (savedConfessions.length === 0) {

                container.innerHTML = '<div class="empty-state">No confessions saved yet.<br>Go to CONFESSIONS to begin.</div>';

                return;

            }

           

            container.innerHTML = savedConfessions.map(c =>

                '<div class="saved-item">' +

                    '<div class="saved-header">' +

                        '<span>' + c.date + '</span>' +

                        '<span>ID: ' + c.id.toString().slice(-6) + '</span>' +

                    '</div>' +

                    '<div class="saved-sin">SIN: ' + c.sin + '</div>' +

                    '<div style="font-size: 11px; color: #404040; margin-bottom: 10px; font-style: italic;">' +

                        c.verse + ' ' + c.reference +

                    '</div>' +

                    '<div class="saved-text">' + escapeHtml(c.text) + '</div>' +

                    '<button class="delete-btn" onclick="deleteConfession(' + c.id + ')">DELETE</button>' +

                '</div>'

            ).join('');

        }

       

        function deleteConfession(id) {

            if (confirm('Are you sure you want to delete this confession?')) {

                savedConfessions = savedConfessions.filter(c => c.id !== id);

                localStorage.setItem('confessions', JSON.stringify(savedConfessions));

                renderSavedConfessions();

                showNotification('DELETED', 'Confession removed from archive.');

            }

        }

       

        function showNotification(title, message, isError) {

            const notif = document.getElementById('notification');

            document.getElementById('notif-title').textContent = title;

            document.getElementById('notif-message').textContent = message;

           

            if (isError) {

                notif.classList.add('error');

            } else {

                notif.classList.remove('error');

            }

           

            notif.classList.add('show');

           

            setTimeout(function() {

                hideNotification();

            }, 4000);

        }

       

        function hideNotification() {

            document.getElementById('notification').classList.remove('show');

        }

       

        function escapeHtml(text) {

            const div = document.createElement('div');

            div.textContent = text;

            return div.innerHTML;

        }

            document.addEventListener('DOMContentLoaded', runDiagnostics);

            runDiagnostics();

        }

       

        window.addEventListener('load', runDiagnostics);

       

        document.addEventListener('keydown', function(e) {

            if (e.key === 'Escape') {

                    document.getElementById('startup').style.display === 'none') {

                    showScreen('main-menu');

                }

            }

        });
